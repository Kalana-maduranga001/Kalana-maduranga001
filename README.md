import React, { useState, useEffect } from 'react';
import { Eye, TrendingUp, Clock } from 'lucide-react';

const ProfileViewCounter = () => {
  const [viewCount, setViewCount] = useState(0);
  const [isLoading, setIsLoading] = useState(true);
  const [lastUpdated, setLastUpdated] = useState(null);

  useEffect(() => {
    // Load and update view count
    const initializeViews = async () => {
      try {
        // Get stored views data
        const result = await window.storage.get('profile-views-24h');
        
        let viewsData = {
          views: [],
          count: 0
        };

        if (result) {
          viewsData = JSON.parse(result.value);
        }

        // Current timestamp
        const now = Date.now();
        const twentyFourHoursAgo = now - (24 * 60 * 60 * 1000);

        // Filter views from last 24 hours
        const recentViews = viewsData.views.filter(timestamp => timestamp > twentyFourHoursAgo);

        // Add current view
        recentViews.push(now);

        // Update storage
        const updatedData = {
          views: recentViews,
          count: recentViews.length
        };

        await window.storage.set('profile-views-24h', JSON.stringify(updatedData));

        setViewCount(recentViews.length);
        setLastUpdated(new Date(now));
        setIsLoading(false);
      } catch (error) {
        console.error('Error managing views:', error);
        // If storage fails, just show count as 1
        setViewCount(1);
        setIsLoading(false);
      }
    };

    initializeViews();
  }, []);

  const formatTime = (date) => {
    if (!date) return '';
    return date.toLocaleTimeString('en-US', { 
      hour: '2-digit', 
      minute: '2-digit',
      hour12: true 
    });
  };

  if (isLoading) {
    return (
      <div className="flex items-center justify-center p-8 bg-gradient-to-br from-gray-900 to-gray-800 rounded-lg">
        <div className="animate-pulse flex items-center gap-3">
          <Eye className="w-6 h-6 text-blue-400" />
          <span className="text-gray-300">Loading views...</span>
        </div>
      </div>
    );
  }

  return (
    <div className="w-full max-w-2xl mx-auto p-6">
      <div className="bg-gradient-to-br from-gray-900 via-gray-800 to-gray-900 rounded-2xl shadow-2xl border border-gray-700 overflow-hidden">
        {/* Header */}
        <div className="bg-gradient-to-r from-blue-600 to-purple-600 p-4">
          <h2 className="text-2xl font-bold text-white flex items-center gap-2">
            <Eye className="w-7 h-7" />
            Profile Views
          </h2>
        </div>

        {/* Main Content */}
        <div className="p-8">
          <div className="flex items-center justify-center gap-6 mb-6">
            <div className="relative">
              <div className="absolute inset-0 bg-blue-500 blur-xl opacity-30 rounded-full"></div>
              <div className="relative bg-gradient-to-br from-blue-500 to-purple-600 rounded-full p-6">
                <Eye className="w-12 h-12 text-white" />
              </div>
            </div>
            
            <div>
              <div className="text-6xl font-bold bg-gradient-to-r from-blue-400 to-purple-400 bg-clip-text text-transparent">
                {viewCount.toLocaleString()}
              </div>
              <div className="text-gray-400 text-sm mt-1 flex items-center gap-2">
                <Clock className="w-4 h-4" />
                Last 24 Hours
              </div>
            </div>
          </div>

          {/* Stats */}
          <div className="grid grid-cols-2 gap-4 mt-6">
            <div className="bg-gray-800/50 rounded-lg p-4 border border-gray-700">
              <div className="flex items-center gap-2 text-gray-400 text-sm mb-1">
                <TrendingUp className="w-4 h-4" />
                Status
              </div>
              <div className="text-green-400 font-semibold">Active</div>
            </div>
            
            <div className="bg-gray-800/50 rounded-lg p-4 border border-gray-700">
              <div className="flex items-center gap-2 text-gray-400 text-sm mb-1">
                <Clock className="w-4 h-4" />
                Updated
              </div>
              <div className="text-blue-400 font-semibold">
                {lastUpdated ? formatTime(lastUpdated) : 'Just now'}
              </div>
            </div>
          </div>

          {/* Info */}
          <div className="mt-6 p-4 bg-blue-900/20 border border-blue-800/30 rounded-lg">
            <p className="text-sm text-gray-300 text-center">
              üí° This counter tracks unique profile views in the last 24 hours using browser storage
            </p>
          </div>
        </div>
      </div>

      {/* Implementation Note */}
      <div className="mt-6 p-4 bg-gray-800/50 rounded-lg border border-gray-700">
        <h3 className="text-white font-semibold mb-2 flex items-center gap-2">
          üìù How to add to your GitHub README:
        </h3>
        <div className="text-sm text-gray-300 space-y-2">
          <p>Since GitHub READMEs are static, you'll need to use one of these methods:</p>
          <ul className="list-disc list-inside space-y-1 ml-2">
            <li><strong>GitHub Actions:</strong> Create a workflow that updates the README with view counts</li>
            <li><strong>External API Service:</strong> Use services like hits.sh or visitor-badge</li>
            <li><strong>Dynamic Badge:</strong> Add this to your README:</li>
          </ul>
          <code className="block bg-gray-900 p-3 rounded text-green-400 text-xs overflow-x-auto mt-2">
            {`![Profile Views](https://komarev.com/ghpvc/?username=Kalana-maduranga001&color=blue&style=flat-square&label=Profile+Views)`}
          </code>
        </div>
      </div>
    </div>
  );
};

export default ProfileViewCounter;
